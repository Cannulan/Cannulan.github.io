这是最关键的一步。你只需要修改两样东西：**API 基地址** 和 **API Key**。

python
import os
from openai import OpenAI

# ========== 配置区域：只需修改这里 ==========
# 1. 你的千聚API Key
api_key = "sk-xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx"  # 替换成你在千聚后台获取的Key

# 2. 核心！修改API基地址为千聚的地址
base_url = "https://www.qianjuai.com/v1"  # 就是这个地址
# =============================================

# 初始化客户端
client = OpenAI(
    api_key=api_key,
    base_url=base_url
)

# 构建请求信息
model_name = "deepseek-chat"  # DeepSeek的模型ID，你也可以用deepseek-r1等

messages = [
    {"role": "system", "content": "你是一个专业的AI助手，回答要精准、简洁。"},
    {"role": "user", "content": "给我用Python写一个斐波那契数列的函数。"}
]

# 发送请求，获取流式输出（推荐，体验更好）
print(f"正在调用 {model_name} 模型...\n")
try:
    stream = client.chat.completions.create(
        model=model_name,
        messages=messages,
        stream=True,  # 开启流式输出
        max_tokens=1024
    )

    for chunk in stream:
        if chunk.choices[0].delta.content is not None:
            print(chunk.choices[0].delta.content, end="")

except Exception as e:
    print(f"请求失败: {e}")

*   **流式输出**：`stream=True` 让你能像 ChatGPT 一样一个字一个字看到回复，体验更流畅。
