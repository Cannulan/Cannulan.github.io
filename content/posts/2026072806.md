java
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;
import java.time.Duration;
import java.util.Arrays;

public class QianJuGPTExample {

    // 千聚api的API端点，直接Base URL
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    // 你在千聚api控制台创建的API Key
    private static final String API_KEY = "sk-你的千聚api密钥";

    public static void main(String[] args) {
        // 1. 构建OpenAiService客户端，设置超时时间
        OpenAiService service = new OpenAiService(BASE_URL, API_KEY, Duration.ofSeconds(60));

        // 2. 构建聊天请求，指定模型为GPT-5.2
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-5.2-turbo") // 千聚api支持的模型名称
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一位资深Java架构师，擅长用简洁的代码解决问题。"),
                        new ChatMessage("user", "用Java写一个单例模式的懒汉式实现，考虑线程安全。")
                ))
                .maxTokens(1024)
                .temperature(0.7)
                .build();

        // 3. 发送请求并打印结果
        service.createChatCompletion(request)
                .getChoices()
                .forEach(choice -> {
                    System.out.println("千聚api响应内容：");
                    System.out.println(choice.getMessage().getContent());
                });

        // 4. 关闭服务，释放资源
        service.shutdown();
    }
}

### 2. 设置客户端：把Base URL指向千聚api

这是最核心的改动：

`BASE_URL`必须用千聚api的端点`https://www.qianjuai.com/v1`，这是国内直连的关键。如果你原来用的是OpenAI官方地址`https://api.openai.com/v1`，把这一行改掉，别的什么都不用动。

### 3. 指定模型：千聚api支持GPT-5.2

请求里设置模型时，我用了`gpt-5.2-turbo`。这是[千聚api聚合平台](https://www.qianjuai.com/)上GPT-5.2的模型标识。实际使用中，你可以去千聚api官网查看最新的模型列表，支持的模型有500多个，OpenAI、Claude、Gemini、DeepSeek应有尽有。

### 4. 处理响应：流式与非流式

上面示例用的是**非流式**调用，适合简单对话场景。如果你要实时流式输出（比如做聊天机器人），用`streamChatCompletion`方法，代码也很直观：

## 分步调试：遇到报错怎么办

如果你跟着代码敲完，运行报错了，别慌。我总结了三个最常见的问题和解决方案，挨个排查就行。

**问题1：连接超时或Network is unreachable**

原因：DNS解析或防火墙拦截了千聚api的域名。  
解决：检查你能否访问`www.qianjuai.com`。能访问，则检查代码中的`BASE_URL`是否写错。不能访问，检查系统代理设置或看看公司网络是否有限制。千聚api的域名在国内可直接访问，不需要任何梯子。

**问题2：401 Unauthorized**

原因：API Key无效或错误。  
解决：去千聚api控制台重新创建API Key，复制的时候小心别漏字符。粘贴到代码里后，记得前后不要有空格。

**问题3：模型名称错误**

原因：你用的模型标识不是千聚api支持的。  
解决：访问千聚api官网的模型列表页面，找到GPT-5.2的确切模型名称。有时是`gpt-5.2`，有时是`gpt-5.2-turbo`，以官方最新列表为准。
