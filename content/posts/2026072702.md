用代码说话最直接。假设你手里有一个千聚的 API Key，你要做的就是在你的 Node.js 项目里，安装 `openai` 这个 npm 包（如果还没装），然后写这几行：

// 1. 安装包 npm install openai --save
// 2. 引入并配置
import OpenAI from 'openai';

const doubao = new OpenAI({
  baseURL: "https://www.qianjuai.com/v1", // 关键：换成千聚的地址
  apiKey: "your-qianjuai-api-key",     // 注意：你在这里填的是千聚给你的Key，不是豆包官方的
});

async function talkToDoubao() {
  const stream = await doubao.chat.completions.create({
    model: "doubao-pro-32k", // 模型名：直接用豆包的名字即可
    messages: [
      { role: "system", content: "你是豆包，一个乐于助人的助手。" },
      { role: "user", content: "用一句诗形容今天的天气很好" },
    ],
    stream: true, // 支持流式输出
  });

  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || "");
  }
}

talkToDoubao();
