const OpenAI = require('openai');

// 2. 创建OpenAI客户端，只改base_url和api_key
const openai = new OpenAI({
  baseURL: 'https://www.qianjuai.com/v1',
  apiKey: '你的千聚API Key', // 去注册后获取
});

async function main() {
  try {
    const completion = await openai.chat.completions.create({
      model: 'gpt-4.1-nano', // 直接调用最新模型
      messages: [
        { role: 'system', content: '你是一个高效的Node.js代码助手。' },
        { role: 'user', content: '帮我写一个快速排序函数，用TypeScript实现。' },
      ],
      max_tokens: 2048,
      temperature: 0.7,
    });

    console.log('AI回复：', completion.choices[0].message.content);
  } catch (error) {
    console.error('请求失败：', error.message);
  }
}

main();
