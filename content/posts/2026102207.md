### 技巧一：设置 `max_completion_tokens` 来限制无脑思考

R1 是一个思维链模型，它会在给出最终答案前，先输出一大段“思考过程”。这个思考过程本身也是要消耗 Token 的。

> **省钱逻辑**：对于简单问题（如“1+1等于几”），你不需要它思考 1000 个 Token。

在千聚接入的 DeepSeek R1 模型中，你可以通过一个巧妙的方式来限制它的“思考预算”。

创建一个 `optimized-request.js` 文件：

javascript
const OpenAI = require('openai');

const client = new OpenAI({
  baseURL: 'https://www.qianjuai.com/v1',
  apiKey: '你的千聚API Key'
});

async function cheapAsk(question, taskType = 'simple') {
  // 核心省钱逻辑 1: 根据任务复杂度控制总输出Token
  let maxTokens = 1024; // 简单任务
  if (taskType === 'complex') {
    maxTokens = 4096; // 复杂推理任务
  }

  const response = await client.chat.completions.create({
    model: 'deepseek-r1',
    messages: [
      { role: 'user', content: question }
    ],
    max_tokens: maxTokens, // 这是最关键的一行！
    stream: false
  });

  const usage = response.usage;
  console.log(`本次请求消耗:
  输入 Token: ${usage.prompt_tokens}
  输出 Token: ${usage.completion_tokens}
  总 Token: ${usage.total_tokens}
  预估费用: \$ ${(usage.total_tokens / 1000000 * 1).toFixed(6)} (基于千聚约1元/1M Tokens的费率)`);

  return response.choices[0]?.message?.content;
}

// 测试：一个简单问题
cheapAsk('Python里怎么打印Hello World？', 'simple').then(reply => {
  console.log('回答:', reply);
});


**效果**：当你设定 `max_tokens: 1024` 时，R1 模型会自动压缩它的内部思考长度。对于简单问题，它不再会输出几千字的内心戏，而是直奔主题。这意味着你少付了很多“思考税”。

### 技巧三：对话压缩与向量召回

这是进阶用法，适合构建聊天机器人。

假设用户问了10个问题，你如果每次把10轮的对话历史都传给API，成本会线性增加。

**正确姿势**：
1.  只保留最近的2-3轮对话。
2.  将历史对话的关键信息，通过向量摘要提取出来，作为 `System Prompt` 的一部分传入。

创建 `conversation-saver.js`：

javascript
const OpenAI = require('openai');

const client = new OpenAI({
  baseURL: 'https://www.qianjuai.com/v1',
  apiKey: '你的千聚限时特价API Key' // 使用特价分组
});

async function smartChat(conversationHistory, newQuestion) {
  // 核心省钱逻辑 2: 只保留最近的几条消息
  const recentHistory = conversationHistory.slice(-4); // 只保留最近两轮对话

  // 构建精简的system message
  const systemMsg = {
    role: 'system',
    content: '你是一个助手。记住，如果用户只是问好，请简短回答。' +
             '如果问题需要复杂推理，务必使用中文回复。' +
             '请尽量精简输出，避免不必要的思考过程。'
  };

  const messages = [
    systemMsg,
    ...recentHistory,
    { role: 'user', content: newQuestion }
  ];

  const response = await client.chat.completions.create({
    model: 'deepseek-r1',
    messages: messages,
    max_tokens: 2048,
    stream: false
  });

  return response.choices[0]?.message?.content;
}

// 模拟一个长对话
let chatHistory = [
  { role: 'user', content: '你好' },
  { role: 'assistant', content: '你好！' },
  { role: 'user', content: '解释一下量子纠缠' },
  { role: 'assistant', content: '量子纠缠是...' },
  { role: 'user', content: '帮我写个React组件' },
];

async function run() {
  const answer = await smartChat(chatHistory, '请优化我刚才那个React组件');
  console.log('优化方案:', answer);
  console.log('注意：我们只发送了极少的历史消息，但保留了正确的对话上下文。');
}

run();


**效果**：通过只传递必要的上下文，大幅削减每次请求的输入 Token 数量。很多对话场景下，输入 Token 的成本比输出 Token 还高。压减历史等于直接从源头省钱。
