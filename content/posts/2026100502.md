java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.Arrays;

public class ErnieBotDemo {
    public static void main(String[] args) {
        // 核心：只改这一个东西
        String apiKey = "sk-你的千聚Key";
        String baseUrl = "https://www.qianjuai.com/v1";

        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(60));
        // 设置自定义base_url（需要修改源码或在构建时指定）
        // 推荐直接通过环境变量或application配置
        System.setProperty("openai.base.url", baseUrl);

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("ernie-4.0-8k") // 指定文心一言4.0模型
                .messages(Arrays.asList(
                        new ChatMessage("user", "请用一句话介绍北京")
                ))
                .maxTokens(200)
                .build();

        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
