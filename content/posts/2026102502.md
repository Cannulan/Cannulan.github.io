o3-mini低代码接入Java示例——代码库详解

为了让你真正上手，我整理了一个现成代码库的骨架。你可以直接抄去用。

**核心工具类**

java
public class AiAssistant {
    private final OpenAiApi api;
    private final String apiKey;

    public AiAssistant(String apiKey) {
        this.apiKey = apiKey;
        OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(30, TimeUnit.SECONDS)
            .readTimeout(60, TimeUnit.SECONDS)
            .build();

        this.api = new Retrofit.Builder()
            .baseUrl("https://www.qianjuai.com/v1")
            .client(client)
            .addCallAdapterFactory(RxJava2CallAdapterFactory.create())
            .addConverterFactory(GsonConverterFactory.create())
            .build()
            .create(OpenAiApi.class);
    }

    public String ask(String model, String prompt) {
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model(model)
            .messages(List.of(new ChatMessage("user", prompt)))
            .build();

        return api.createChatCompletion(request)
            .blockingGet()
            .getChoices()
            .get(0)
            .getMessage()
            .getContent();
    }
}
