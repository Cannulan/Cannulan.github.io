        // 3. 构建请求，这里用文心一言 ERNIE-4.0 模型
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("ernie-4.0-8k")
            .messages(Arrays.asList(
                new ChatMessage("system", "你是一个专业的编程助手。"),
                new ChatMessage("user", "用 Java 写一个快速排序算法。")
