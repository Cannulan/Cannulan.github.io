### 3.2 编写后端逻辑代码

这里以最简单的Python + Flask为例。假设你已经部署了一个小型服务器（或用云函数）。

核心代码只需要改一个地方——让你的请求指向千聚的API地址：

python
import requests
from flask import Flask, request, jsonify

app = Flask(__name__)

# 千聚API基础地址
API_BASE = "https://www.qianjuai.com/v1"
API_KEY = "sk-你的千聚API密钥"  # 替换成你刚拿到的Key

@app.route('/webhook', methods=['POST'])
def handle_webhook():
    data = request.json
    user_message = data['text']['content']
    
    # 调用千聚的API
    headers = {"Authorization": f"Bearer {API_KEY}"}
    payload = {
        "model": "gpt-4o-mini",  # 你可以随意切换
        "messages": [{"role": "user", "content": user_message}]
    }
    response = requests.post(f"{API_BASE}/chat/completions", 
                             json=payload, 
                             headers=headers)
    
    ai_reply = response.json()['choices'][0]['message']['content']
    
    # 返回给钉钉
    return jsonify({
        "msgtype": "text",
        "text": {"content": ai_reply}
    })

if __name__ == '__main__':
    app.run(port=5000)
