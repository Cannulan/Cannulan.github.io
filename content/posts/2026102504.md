import okhttp3.*;
import com.google.gson.JsonArray;
import com.google.gson.JsonObject;
import java.io.IOException;

public class QianjuAiErnieDemo {

    // **重要：请替换成你的千聚API Key**
    private static final String API_KEY = "你的千聚API_KEY";
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    // 指定模型，千聚会自动映射到ERNIE-Bot 4.0
    private static final String MODEL_NAME = "ernie-bot-4.0";

    private static final OkHttpClient client = new OkHttpClient();
    private static final MediaType JSON = MediaType.get("application/json; charset=utf-8");

    public static String generateResponse(String userMessage) throws IOException {
        // 构建请求体
        JsonArray messages = new JsonArray();
        JsonObject userMessageObj = new JsonObject();
        userMessageObj.addProperty("role", "user");
        userMessageObj.addProperty("content", userMessage);
        messages.add(userMessageObj);

        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", MODEL_NAME);
        requestBody.add("messages", messages);

        // 发送POST请求
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .post(RequestBody.create(requestBody.toString(), JSON))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("Unexpected code " + response.code());
            }

            String responseBody = response.body().string();
            JsonObject jsonObject = new com.google.gson.Gson().fromJson(responseBody, JsonObject.class);

            // 提取回复内容
            String content = jsonObject.getAsJsonArray("choices")
                    .get(0).getAsJsonObject()
                    .getAsJsonObject("message")
                    .get("content")
                    .getAsString();
            return content;
        }
    }

    public static void main(String[] args) throws IOException {
        // 测试调用
        String result = generateResponse("请用中文告诉我如何用Java实现数据库连接池？");
        System.out.println("ERNIE回复：\n" + result);
    }
}
