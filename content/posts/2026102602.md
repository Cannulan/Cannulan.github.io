**2. 核心调用代码**

java
import okhttp3.*;
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;
import com.alibaba.fastjson.JSONArray;
import java.io.IOException;

public class QwenVLWithAggregator {

    private static final String API_KEY = "sk-你的[千聚api聚合站](https://www.qianjuai.com/)Key";
    // 重点！！！ 把baseUrl从阿里那边指向千聚
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient().newBuilder()
                .build();

        // 构建请求体，完全兼容OpenAI标准格式，传图支持URL或Base64
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "qwen-vl-plus"); // 指定模型为千问VL

        JSONArray messages = new JSONArray();
        JSONObject message = new JSONObject();
        message.put("role", "user");

        JSONArray content = new JSONArray();
        // 文本部分
        JSONObject textPart = new JSONObject();
        textPart.put("type", "text");
        textPart.put("text", "请描述这张图片里有什么内容，并包含什么文字记录。");

        // 图片部分（这里用一张互联网公开的测试图片URL）
        JSONObject imagePart = new JSONObject();
        imagePart.put("type", "image_url");
        JSONObject imageUrl = new JSONObject();
        // 你可以替换成你自己的图片URL或Base64
        imageUrl.put("url", "https://www.qianjuai.com/register");
        imagePart.put("image_url", imageUrl);

        content.add(textPart);
        content.add(imagePart);
        message.put("content", content);
        messages.add(message);

        requestBody.put("messages", messages);
        requestBody.put("max_tokens", 1000); // 控制输出长度

        // 构建HTTP请求
        MediaType mediaType = MediaType.parse("application/json");
        RequestBody body = RequestBody.create(mediaType, requestBody.toJSONString());
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions") // 标准的/chat/completions接口
                .method("POST", body)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .build();

        // 发送请求并获取流式响应
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseStr = response.body().string();
                JSONObject jsonResponse = JSON.parseObject(responseStr);

                // 提取模型返回的文字描述
                String contentResult = jsonResponse
                    .getJSONArray("choices")
                    .getJSONObject(0)
                    .getJSONObject("message")
                    .getString("content");

                System.out.println("=== Qwen-VL 识别结果 ===");
                System.out.println(contentResult);
            } else {
                System.out.println("请求失败: " + response.code() + " " + response.message());
            }
        }
    }
}
