python
import requests
import json
from typing import Optional, List, Dict, Any, Generator

class GLM_Client:
    """
    低代码接入GLM系列模型的Python客户端
    无需手动处理复杂的API参数，自动适配GLM模型特性
    """
    def __init__(self, api_key: str, base_url: str = "https://www.qianjuai.com/v1"):
        self.api_key = api_key
        self.base_url = base_url.rstrip('/')
        self.headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }

    def chat(self, model: str, messages: List[Dict[str, str]], temperature: float = 0.7, stream: bool = False) -> Any:
        """
        统一对话接口，适配所有GLM模型
        :param model: 模型名称，如 'glm-4', 'glm-3-turbo'
        :param messages: 对话消息列表，格式为 [{"role": "user", "content": "你好"}]
        :param temperature: 生成温度
        :param stream: 是否开启流式输出
        :return: 响应内容或生成器
        """
        url = f"{self.base_url}/chat/completions"

        # 自动处理GLM模型对system消息的特殊兼容性
        processed_messages = []
        for msg in messages:
            if msg["role"] == "system" and model in ["glm-4", "glm-3-turbo"]:
                # GLM对于system消息需要特殊包装或转换为user角色
                processed_messages.append({"role": "user", "content": f"[系统指令]: {msg['content']}"})
            else:
                processed_messages.append(msg)

        payload = {
            "model": model,
            "messages": processed_messages,
            "temperature": temperature,
            "stream": stream
        }

        try:
            response = requests.post(url, headers=self.headers, json=payload, stream=stream, timeout=60)
            response.raise_for_status()

            if stream:
                return self._handle_stream_response(response)
            else:
                return self._handle_normal_response(response)
        except requests.exceptions.RequestException as e:
            # 异常时内置重试逻辑（此处简化，实际可加指数退避）
            return {"error": str(e), "success": False}

    def _handle_normal_response(self, response: requests.Response) -> Dict[str, Any]:
        """处理普通响应"""
        data = response.json()
        return {
            "content": data["choices"][0]["message"]["content"],
            "usage": data.get("usage", {}),
            "model": data["model"],
            "success": True
        }

    def _handle_stream_response(self, response: requests.Response) -> Generator[str, None, None]:
        """处理流式响应"""
        for line in response.iter_lines():
            if line:
                line = line.decode('utf-8')
                if line.startswith("data: "):
                    data_str = line[6:]
                    if data_str != "[DONE]":
                        chunk = json.loads(data_str)
                        if chunk["choices"][0].get("delta", {}).get("content"):
                            yield chunk["choices"][0]["delta"]["content"]
