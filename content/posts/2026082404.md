# 注意：请将下面的路径改为你电脑上实际存图片的路径

        {
            "role": "user",
            "content": [
                {
                    "type": "text",
                    "text": "请描述一下这张图片里的主要内容是什么？",
                },
                {
                    "type": "image_url",
                    "image_url": {
                        "url": f"data:image/jpeg;base64,{base64_image}"
                    },
                },
            ],
        }
    ],
    max_tokens=300,
)

# 4. 打印模型的回复
