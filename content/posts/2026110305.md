3.  **安装必要库**：在电脑上执行 `pip install dingtalk-stream openai`。

python
import dingtalk_stream
from openai import OpenAI

def setup_dingtalk_bot():
    credential = dingtalk_stream.DingTalkStreamCredentials(
        client_id='你的钉钉ClientId',
        client_secret='你的钉钉ClientSecret'
    )

    # 配置千聚中转站
    client = OpenAI(
        api_key="你的千聚API_Key",
        base_url="https://www.qianjuai.com/v1"
    )

    client_ding = dingtalk_stream.DingTalkStreamClient(credential)

    @client_ding.chat_bot_message_handler
    def handle_bot_message(msg: dingtalk_stream.ChatBotMessage):
        # 调用千聚的Grok3模型
        response = client.chat.completions.create(
            model="grok3",
            messages=[
                {"role": "system", "content": "你是一个企业智能助理，请专业、简洁地回答。"},
                {"role": "user", "content": msg.text.content}
            ]
        )

        reply_text = response.choices[0].message.content
        msg.reply_text(reply_text)

    client_ding.start_forever()

if __name__ == '__main__':
    setup_dingtalk_bot()
