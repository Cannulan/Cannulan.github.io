import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

/**

 * 新手100%成功，免翻墙，复制粘贴修改API Key即可运行

public class MistralDemo {

    // ⚠️ 关键配置：远端服务器的地址（OpenAI兼容格式）
    // [千聚api中转站](https://www.qianjuai.com/)的地址，直接固定，一行搞定
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    // ⚠️ 这也是关键：你在[千聚api中转站](https://www.qianjuai.com/)后台创建的API Key
    // 我这里写的是替换提示，你要换成你自己的真实Key
    private static final String API_KEY = "sk-xxxxxxxxxxxxxxxx"; // 替换成你的Key

    // Mistral模型名称：推荐使用经典的 mistral-large-latest
    private static final String MODEL_NAME = "mistral-large-latest";

    public static void main(String[] args) {
        System.out.println("=== [千聚api中转站](https://www.qianjuai.com/) Mistral 接入Demo启动 ===");

        // 1. 创建OpenAiService实例
        //    关键点：传入自定义的BASE_URL，指向[千聚api中转站](https://www.qianjuai.com/)，而不是官方的api.openai.com
        //    这样就实现了“免翻墙、国内直连”
        OpenAiService service = new OpenAiService(
                API_KEY,
                Duration.ofSeconds(60) // 超时时间，60秒
        );

        // 2. 构建聊天消息列表
        List<ChatMessage> messages = new ArrayList<>();
        // 系统Prompt：告诉Mistral你的身份和风格
        messages.add(new ChatMessage("system", "你是一个技术领域的资深专家，说话简洁明了，回答中文问题。"));
        // 用户问题：你想问什么就在这里写
        messages.add(new ChatMessage("user", "请用Java写一个快速排序算法的示例代码，并解释时间复杂度。"));

        // 3. 构建请求体
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model(MODEL_NAME)
                .messages(messages)
                .temperature(0.7)  // 控制创造力，0到2之间，越小越保守
                .maxTokens(2048)   // 最大输出token数
                .build();

        try {
            // 4. 发起网络请求，获取结果
            //    这一步会自动用你给的BASE_URL和API Key去[千聚api中转站](https://www.qianjuai.com/)，
            //    然后由中转站转发给Mistral大模型，拿到结果后返回。
            System.out.println("正在请求模型 " + MODEL_NAME + " ...");
            var result = service.createChatCompletion(request);

            // 5. 抽取并打印回答
            String response = result.getChoices().get(0).getMessage().getContent();
            System.out.println("=== 模型返回的结果 ===");
            System.out.println(response);

        } catch (Exception e) {
            System.err.println("调用失败，检查网络或API Key是否正确！");
            e.printStackTrace();
        }

        System.out.println("=== Demo结束 ===");
    }
