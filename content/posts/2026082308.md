    model="mistral-large-latest",
    messages=[
        {"role": "system", "content": "你是一个创意助手，你的回答要极其简洁且充满洞见。"},
        {"role": "user", "content": "用一句话解释为什么流式输出对AI交互至关重要。"}

    stream=True  # 注意这个参数

    content = chunk.choices[0].delta.content
    if content:
        print(content, end="", flush=True)
